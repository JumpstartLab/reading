def to_csv_string
  # We shouldn't rely on the hash order event tho it's a Ruby 1.9 feature
  CSV.generate do |csv|
    # add headers
    csv << EventReporter::PRINT_HEADERS
    self.each do |row|
      csv << row.values_at(*EventReporter::PRINT_ATTRIBUTES_ORDER) if row.respond_to?(:values_at)
    end
  end
end
